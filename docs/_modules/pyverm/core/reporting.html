[docs]    def to_txt(self, filename):
        """
        Saves current report to the given filename as txt.

        :param filename: string (path with filename)
        """
        file = open(filename, "w", encoding="UTF-8")
        file.write(self._report)
        file.close()


    def __str__(self):
        return self._report
